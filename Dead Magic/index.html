namespace Player.Attack
{
    public class AttackManager : NetworkBehaviour
    {
        public enum Spells
        {
            Basic,
            Fireball,
            Bolt,
            Arcane
        }

        [Header("Spell Prefabs")]
        [SerializeField] private GameObject[] spellObject;

        [Header("Spell Indicators")]
        [SerializeField] private RawImage fireballIndicator;
        [SerializeField] private Canvas boltIndicator;
        [SerializeField] private Canvas arcaneIndicator;

        [Header("Other")]
        [SerializeField] private LayerMask groundMask;

        private readonly Dictionary _spellDictionary = new();
        private readonly Dictionary _spellCooldowns = new();
        private readonly Dictionary _indicators = new();

        private Spell _selectedSpell;       // currently selected spell SO
        private Spells _currentSpell;       // currently selected spell enum
        private Spell _castedSpell;         // spell currently being cast
        private NetworkObject castedSpell;  // cast spell object

        private PlayerStats _playerStats;
        private PlayerMovement _playerMovement;
        private AniManager _playerAnimator;
        private Camera _camera;
        private ulong _playerId;

        private int _lastHeldSpell;
        private bool _isHoldingSpell;

        public bool cd;
        public bool fireballCd;

        #region Unity Callbacks

        private void Start()
        {
            _playerId = NetworkManager.Singleton.LocalClientId;
            groundMask = LayerMask.GetMask("Ground");

            _playerStats = GetComponent();
            _playerMovement = GetComponent();
            _playerAnimator = GetComponent();
            _camera = Camera.main;

            InitializeSpells();
            InitializeIndicators();

            _selectedSpell = SetSpell(Spells.Basic);
        }

        private void Update()
        {
            if (!IsOwner) return;

            HandleSpellInput();
            HandleCasting();

            if (!_isHoldingSpell && InputHandler.Instance.attackTriggered)
            {
                SetSpell(Spells.Basic);
                HandleSpellRelease();
            }
        }

        #endregion

        #region Initialization

        private void InitializeSpells()
        {
            foreach (var spell in Resources.LoadAll("Spells"))
            {
                if (Enum.TryParse(spell.name, out Spells spellEnum))
                {
                    _spellDictionary[spellEnum] = spell;
                    _spellCooldowns[spellEnum] = 0f;
                }
            }
        }

        private void InitializeIndicators()
        {
            _indicators[Spells.Fireball] = fireballIndicator.gameObject;
            _indicators[Spells.Bolt] = boltIndicator.gameObject;
            _indicators[Spells.Arcane] = arcaneIndicator.gameObject;
        }

        #endregion

        #region Spell Selection

        private void HandleSpellInput()
        {
            int currentHeldSpell = InputHandler.Instance.spellIndex;

            if (currentHeldSpell != 0 && currentHeldSpell != _lastHeldSpell)
            {
                switch (currentHeldSpell)
                {
                    case 1: SetSpell(Spells.Fireball); break;
                    case 2: SetSpell(Spells.Bolt); break;
                    case 3: SetSpell(Spells.Arcane); break;
                }

                _isHoldingSpell = true;
                _lastHeldSpell = currentHeldSpell;
            }

            if (_isHoldingSpell && currentHeldSpell == 0)
            {
                HandleSpellRelease();
            }
        }

        private void HandleSpellRelease()
        {
            if (cd || fireballCd || _selectedSpell == null) return;

            if (!Physics.Raycast(_camera.ScreenPointToRay(Input.mousePosition), out var hit, Mathf.Infinity, groundMask)) return;
            if (Time.time <= _spellCooldowns[_currentSpell]) return;

            Vector3 hitPos = hit.point;
            _playerAnimator.ChangeAnimation(_currentSpell.ToString(), layer: 1);
            _castedSpell = _selectedSpell;
            _spellCooldowns[_currentSpell] = Time.time + _selectedSpell.cooldown;

            if (CheckMana() && IsOwner)
            {
                CastSpellRpc(hitPos, (int)_currentSpell, _playerStats.currentMana);
            }

            _isHoldingSpell = false;
            _lastHeldSpell = 0;
            _currentSpell = Spells.Basic;
        }

        public Spell SetSpell(Spells spellName)
        {
            if (!_spellDictionary.TryGetValue(spellName, out var spell) || spell == null)
            {
                Debug.LogWarning("Spell not found: " + spellName);
                return null;
            }

            if (_playerStats.currentMana >= spell.manaCost && !cd)
            {
                _selectedSpell = spell;
                _currentSpell = spellName;
                SetIndicator(spellName);
                UIManager.Instance.UpdateSelectedSpell((int)_currentSpell);
                return spell;
            }

            if (!cd)
            {
                UIManager.Instance.FlashManaBarBackground();
                Debug.Log("Not enough mana to cast " + spell.name);
            }

            _selectedSpell = _spellDictionary[Spells.Basic];
            _currentSpell = Spells.Basic;
            UIManager.Instance.UpdateSelectedSpell((int)_currentSpell);
            SetIndicator(Spells.Basic);

            return _selectedSpell;
        }

        private void SetIndicator(Spells selectedSpell)
        {
            if (cd) return;

            foreach (var indicator in _indicators.Values)
            {
                indicator.SetActive(false);
            }

            if (_indicators.TryGetValue(selectedSpell, out var activeIndicator))
            {
                activeIndicator.SetActive(true);
            }
        }

        #endregion

        #region Spell Casting

        private bool CheckMana()
        {
            if (_playerStats.currentMana >= _castedSpell.manaCost) return true;

            Debug.Log("Not enough mana to cast " + _castedSpell.name);
            SoundManager.Instance.PlaySound2D("NoMana");
            return false;
        }

        [Rpc(SendTo.Server)]
        private void CastSpellRpc(Vector3 pos, int spellIndex, float currentMana)
        {
            if (!_spellDictionary.TryGetValue((Spells)spellIndex, out var spell))
            {
                Debug.LogError($"Server: Could not find spell for index {spellIndex}");
                return;
            }

            _castedSpell = spell;
            currentMana -= _castedSpell.manaCost;

            GameObject spawnPosition = (_castedSpell.areaOfEffect == Spell.AreaOfEffect.Cone ||
                                        _castedSpell.areaOfEffect == Spell.AreaOfEffect.Line ||
                                        _castedSpell.areaOfEffect == Spell.AreaOfEffect.None)
                ? spellObject[0]
                : spellObject[1];

            PlayCastSound(spell);

            if (_castedSpell.areaOfEffect == Spell.AreaOfEffect.None ||
                _castedSpell.areaOfEffect == Spell.AreaOfEffect.Circle)
            {
                castedSpell = NetworkManager.Singleton.SpawnManager.InstantiateAndSpawn(
                    spell.spellPrefab, _playerId, spawnPosition.transform.position, Quaternion.identity);

                if (castedSpell != null)
                    castedSpell.GetComponent().useGravity = false;
            }

            if (spell.areaOfEffect != Spell.AreaOfEffect.None) fireballCd = true;

            StartCoroutine(CastSpell(spell, pos));
            SendInfoToOwnerRpc(currentMana, spell.castTime, pos);
        }

        private void PlayCastSound(Spell spell)
        {
            switch (spell.areaOfEffect)
            {
                case Spell.AreaOfEffect.Cone: SoundManager.Instance.PlaySound3D("FrostWave", transform.position); break;
                case Spell.AreaOfEffect.Circle: SoundManager.Instance.PlaySound3D("FireballWindup", transform.position); break;
            }
        }

        [Rpc(SendTo.Owner)]
        private void SendInfoToOwnerRpc(float mana, float castTime, Vector3 pos)
        {
            _playerStats.currentMana = mana;
            StartCoroutine(SetRotationLocal(castTime, pos));
        }

        private IEnumerator SetRotationLocal(float castTime, Vector3 pos)
        {
            float endTime = Time.time + castTime;
            while (Time.time < endTime)
            {
                Quaternion targetRotation = Quaternion.LookRotation(pos - transform.position).normalized;
                transform.rotation = Quaternion.Slerp(transform.rotation, targetRotation, castTime * Time.deltaTime * 50f);
                yield return new WaitForFixedUpdate();
            }
        }

        private IEnumerator CastSpell(Spell spell, Vector3 pos)
        {
            foreach (var indicator in _indicators.Values)
                indicator.SetActive(false);

            cd = true;
            _playerMovement.canMove.Value = false;
            _isHoldingSpell = false;

            yield return new WaitForSeconds(spell.castTime);

            _playerMovement.canMove.Value = true;
            _playerAnimator.ChangeAnimation("Idle", layer: 1);

            if (spell.areaOfEffect == Spell.AreaOfEffect.None)
                SpawnBasicAttackRpc(pos);
            else
                SpawnAoeAttackRpc(pos);

            WaitRpc();
            cd = false;
        }

        [Rpc(SendTo.Server)]
        private void SpawnAoeAttackRpc(Vector3 pos)
        {
            SpawnAoeAttack(_castedSpell.areaOfEffect, pos);
        }

        [Rpc(SendTo.Everyone)]
        private void WaitRpc()
        {
            _selectedSpell = SetSpell(Spells.Basic);
        }

        #endregion

        #region Spell Attacks (Basic, AOE)

        [Rpc(SendTo.Server)]
        private void SpawnBasicAttackRpc(Vector3 pos) => StartCoroutine(BasicAttack(pos));

        private IEnumerator BasicAttack(Vector3 pos)
        {
            SoundManager.Instance.PlaySound3D("BasicAttack", transform.position);

            castedSpell.GetComponent().SetCaster(gameObject);
            castedSpell.GetComponent().useGravity = false;
            castedSpell.GetComponent().linearVelocity = (pos - transform.position).normalized * _castedSpell.travelSpeed;

            while (true)
            {
                if (castedSpell == null || !castedSpell.IsSpawned || spellObject[0] == null)
                    yield break;

                float distance = Vector3.Distance(spellObject[0].transform.position, castedSpell.transform.position);
                if (distance > _castedSpell.range)
                {
                    castedSpell.Despawn();
                    yield break;
                }

                yield return null;
            }
        }

private void SpawnAoeAttack(Spell.AreaOfEffect area, Vector3 targetPos)
{
    if (_castedSpell == null || _castedSpell.hitboxPrefab == null) return;

    Vector3 scale = Vector3.one;
    Quaternion rotation = Quaternion.identity;
    Vector3 position = transform.position;

    switch (area)
    {
        case Spell.AreaOfEffect.Circle:
            scale = Vector3.one * (_castedSpell.areaOfEffectRadius / 2);
            position = targetPos;
            break;

        case Spell.AreaOfEffect.Line:
            scale = new Vector3(_castedSpell.areaOfEffectRadius / 2, 3, (_castedSpell.range / 4) * 2);
            rotation = Quaternion.LookRotation(targetPos - transform.position);
            position += rotation * Vector3.forward * (((_castedSpell.range / 4) * 2 / 10) * 6);
            break;

        case Spell.AreaOfEffect.Cone:
            scale = new Vector3((_castedSpell.areaOfEffectRadius / 5) * 11, (_castedSpell.range / 15) * 8.5f, 3);
            rotation = Quaternion.LookRotation(targetPos - transform.position);
            rotation = Quaternion.Euler(rotation.eulerAngles.x - 90, rotation.eulerAngles.y + 180, rotation.eulerAngles.z);
            break;
    }

    _castedSpell.hitboxPrefab.transform.localScale = scale;

    NetworkObject hitbox = NetworkManager.Singleton.SpawnManager.InstantiateAndSpawn(
        _castedSpell.hitboxPrefab, _playerId, position, rotation);
    hitbox.GetComponent().SetCaster(gameObject);

    StartCoroutine(HandleAoeDuration(hitbox, _castedSpell.duration));
}

private IEnumerator HandleAoeDuration(NetworkObject hitbox, float duration)
{
    yield return new WaitForSeconds(duration);
    hitbox.Despawn(true);
    fireballCd = false;
}

        #endregion

        #region Indicator Handling

        private void HandleCasting()
        {
            if (!Physics.Raycast(Camera.main.ScreenPointToRay(Input.mousePosition), out var hit, Mathf.Infinity, groundMask)) return;
            Vector3 hitPos = hit.point;

            // Fireball indicator
            if (_indicators.TryGetValue(Spells.Fireball, out var fireball) && fireball.activeSelf)
            {
                var targetIndicator = fireball.transform.Find("TargetIndicator");
                targetIndicator.position = hitPos;
                targetIndicator.GetComponent().localScale = Vector3.one * (_selectedSpell.areaOfEffectRadius / 2);
            }

            // Other indicators
            foreach (var indicator in _indicators)
            {
                if (!_selectedSpell || indicator.Key == Spells.Fireball) continue;

                var rect = indicator.Value.GetComponent();
                switch (_selectedSpell.areaOfEffect)
                {
                    case Spell.AreaOfEffect.Line:
                        rect.localScale = new Vector3(_selectedSpell.areaOfEffectRadius, 3, _selectedSpell.range / 4);
                        break;
                    case Spell.AreaOfEffect.Cone:
                        rect.localScale = new Vector3(_selectedSpell.areaOfEffectRadius, 3, _selectedSpell.range / 3);
                        break;
                    case Spell.AreaOfEffect.Circle:
                        rect.localScale = Vector3.one;
                        rect.sizeDelta = new Vector2(_selectedSpell.range / 2, _selectedSpell.range / 2);
                        break;
                }

                if (indicator.Value.activeSelf)
                {
                    Vector3 dir = (hitPos - indicator.Value.transform.position).normalized;
                    indicator.Value.transform.rotation = Quaternion.Euler(0, Quaternion.LookRotation(dir).eulerAngles.y, 0);
                }
            }
        }

        #endregion

        public Spell GetCastedSpell() => _castedSpell;
    }
}

public class PlayerMovement : NetworkBehaviour
  {
      private PlayerStats _playerStats;
      private AniManager _playerAnimator;
      private AttackManager _attackManager;
      private Vector2 _moveInput;
      private Rigidbody _rb;
      private CinemachineCamera _playerCam;
      public NetworkVariable canMove;

      [SerializeField]
      private float moveSpeed;

      [SerializeField]
      private float rotationSpeed;

      [SerializeField]
      private float maxVel;

      private void Awake()
      {
          canMove.Initialize(this); 
          canMove.Value = true;
      }

      private void Start()
      {
          
          if (!IsOwner)
          {
              GetComponentInChildren().enabled = false;
              enabled = false;
              return;
          }

          _playerCam = FindFirstObjectByType();
          _playerStats = GetComponent();
          _playerAnimator = GetComponent();
          _attackManager = GetComponent();
          _rb = GetComponent();

          _playerCam.Target.TrackingTarget = gameObject.transform;
          _playerCam.Target.LookAtTarget = gameObject.transform;
          _playerStats.OnSpeedChanged += f => moveSpeed *= f;
      }

      public override void OnNetworkSpawn()
      {
          base.OnNetworkSpawn();
          RealRpc();
      }

      [Rpc(SendTo.Server)]
      private void RealRpc()
      {
          
          canMove.Value = true; // reset on respawn
      }


      private void Update()
      {
          maxVel = moveSpeed;
          _moveInput = InputHandler.Instance.moveInput;
          if (_rb.linearVelocity.magnitude < 0.1f && !_attackManager.cd)
          {
              _playerAnimator.ChangeAnimation("Idle", 0.2f);
          }
      }

      private void FixedUpdate()
      {
          if (!canMove.Value) return;
          Move();
      }


      private void Move()
      {
          Vector3 moveDir = new Vector3(_moveInput.x, 0, _moveInput.y);
          moveDir.Normalize();
          if (moveDir != Vector3.zero)
          {
              _playerAnimator.ChangeAnimation("Walking", layer: 0);
              Rotation(moveDir);
              _rb.AddForce(moveDir * moveSpeed, ForceMode.VelocityChange);
              if (_rb.linearVelocity.magnitude > maxVel)
              {
                  _rb.linearVelocity = _rb.linearVelocity.normalized * maxVel;
              }
          }
      }

      private void Rotation(Vector3 moveDir)
      {
          Quaternion targetRotation = Quaternion.LookRotation(moveDir);
          transform.rotation = Quaternion.Slerp(transform.rotation, targetRotation, Time.deltaTime * rotationSpeed);
      }
  }
